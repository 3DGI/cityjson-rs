`cityjson-index` builds a SQLite-backed random-access index over CityJSON
datasets. The POC covers the Rust and Python APIs.


<LanguagePanel language="rust">
<ApiCodeBlock crate="cityjson-index" language="rust" code={`use cityjson_index::{CityIndex, LodSelection, PackageFilter, StorageLayout};

let mut index = CityIndex::open(StorageLayout::SingleFile(dataset.into()), &index_path)?;
CityIndex::reindex(&mut index)?;
let cityobjects = CityIndex::lookup_cityobject_refs(&index, "building-1")?;
let filter = PackageFilter {
    default_lod: LodSelection::Highest,
    ..PackageFilter::default()
};
let packages = CityIndex::read_filtered_packages(&index, &[], &filter)?;`} />
</LanguagePanel>

<LanguagePanel language="python">
<ApiCodeBlock crate="cityjson-index" language="python" code={`from cityjson_index import OpenedIndex, PackageFilter, PackageFilterSummary, LodSelection

with OpenedIndex.open("dataset", ".cityjson-index.sqlite") as index:
    OpenedIndex.reindex(index)
    cityobjects = OpenedIndex.lookup_cityobject_refs(index, "building-1")
    refs = index.package_refs_for_cityobject(cityobjects[0])
    filter = PackageFilter(default_lod=LodSelection.HIGHEST)
    summary = PackageFilterSummary()
    for package in index.read_filtered_packages(refs, filter):
        summary.add(package.report)`} />
</LanguagePanel>
